```python PYTHON
import cohere

co = cohere.Client(
    api_key="AZURE_API_KEY",
    base_url="AZURE_ENDPOINT",
)

documents = [
    "Reimbursing Travel Expenses: Easily manage your travel expenses by submitting them through our finance tool.",
    "Working from Abroad: Coordinate with your manager and ensure your availability during core hours.",
    "Health and Wellness Benefits: We offer gym memberships, on-site yoga classes, and comprehensive health insurance.",
]

# Rerank the documents
results = co.rerank(
    model="YOUR_ENDPOINT_NAME",
    query = "Are there fitness-related benefits?",
    documents=documents,
    top_n=2
)

for result in results.results:
    print(result)
```
