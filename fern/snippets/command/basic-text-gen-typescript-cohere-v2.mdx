```typescript TYPESCRIPT
const { CohereClientV2 } = require('cohere-ai');

const cohere = new CohereClientV2({
  token: 'COHERE_API_KEY',
});

(async () => {
  const response = await cohere.chat({
    model: 'command-r-plus-08-2024',
    messages: [
      {
        role: 'user',
        content: 'Hello world!',
      },
    ],
  });

  console.log(response);
})();
```
