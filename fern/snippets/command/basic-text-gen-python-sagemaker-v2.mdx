```python PYTHON
import cohere

    aws_region="AWS_REGION",
    aws_access_key="AWS_ACCESS_KEY_ID",
    aws_secret_key="AWS_SECRET_ACCESS_KEY",
    aws_session_token="AWS_SESSION_TOKEN",
)

response = co.chat(
    model="YOUR_ENDPOINT_NAME", 
    messages=[{"role": "user", "content": "Hello world!"}]
)

print(response)
```
